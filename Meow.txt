local function flashEffect()
    if not effectEnabled then return end 
    task.spawn(function()
        effectImage.ImageTransparency = 0.6 
        task.wait(0.3)
        for i = 0.6, 1, 0.05 do
            effectImage.ImageTransparency = i
            task.wait(0.02)
        end
        effectImage.ImageTransparency = 1
    end)
end

-- Поиск ближайшего
local function getNearest(range)
    local target, dist = nil, range or math.huge
    local char = LP.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LP and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local d = (char.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude
            if d < dist then target = p.Character; dist = d end
        end
    end
    return target
end

-- ФУНКЦИИ ВИЗУАЛОВ
local function updateVisuals()
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            local char = p.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            
            -- ESP Highlight / Chams
            local hl = char:FindFirstChild("ESPHL")
            if espEnabled or chamsEnabled then
                if not hl then 
                    hl = Instance.new("Highlight", char)
                    hl.Name = "ESPHL"
                end
                hl.Enabled = true
                hl.FillAlpha = chamsEnabled and 0.5 or 0
                hl.OutlineAlpha = espEnabled and 0 or 1
                hl.FillColor = Color3.fromRGB(150, 0, 255)
            elseif hl then 
                hl:Destroy() 
            end

            -- Hitbox
            if hrp then
                if hitboxEnabled then
                    hrp.Size = Vector3.new(hitboxSize, hitboxSize, hitboxSize)
                    hrp.Transparency = 0.7
                    hrp.Color = Color3.fromRGB(255, 0, 0)
                    hrp.CanCollide = false
                else
                    hrp.Size = Vector3.new(2, 2, 1)
                    hrp.Transparency = 1
                end
            end
        end
    end
end

-- Трассеры (RunService)
local tracerLines = {}
RunService.RenderStepped:Connect(function()
    for _, line in pairs(tracerLines) do line:Destroy() end
    table.clear(tracerLines)

    if tracersEnabled then
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= LP and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                local pos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(p.Character.HumanoidRootPart.Position)
                if onScreen then
                    local line = Drawing.new("Line")
                    line.Visible = true
                    line.From = Vector2.new(workspace.CurrentCamera.ViewportSize.X / 2, workspace.CurrentCamera.ViewportSize.Y)
                    line.To = Vector2.new(pos.X, pos.Y)
                    line.Color = Color3.fromRGB(255, 255, 255)
                    line.Thickness = 1
                    table.insert(tracerLines, line)
                end
            end
        end
    end
    updateVisuals()
end)

-- Fly / Noclip логика (без изменений)
local function toggleFlyNoclip()
    local speaker = LP
    local char = speaker.Character
    if not char or not char:FindFirstChild("Humanoid") then return end
    local hum = char.Humanoid
    if flyNoclipEnabled then
        tpwalking = true
        task.spawn(function()
            local hb = RunService.Heartbeat
            while tpwalking and speaker.Character and hum and hum.Parent do
                hb:Wait()
                if hum.MoveDirection.Magnitude > 0 then speaker.Character:TranslateBy(hum.MoveDirection * flySpeeds) end
            end
        end)
        hum:ChangeState(Enum.HumanoidStateType.Swimming)
    else
        tpwalking = false
        hum:ChangeState(Enum.HumanoidStateType.RunningNoPhysics)
    end
end

    end
end)
